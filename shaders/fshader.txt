#version 330 core
                                       
out vec4 color;
uniform sampler2D theTexture;
in vec4 pdc;
in vec2 texCoords;
in float transparency;

void main(){
    color = texture(theTexture, texCoords) * vec4(1.0, 1.0f, 1.0f, transparency);
if (texture(theTexture, texCoords).a < 0.5)
    discard;

};
